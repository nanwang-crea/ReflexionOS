# ReflexionOS v0.0.1.1 增强 MVP 执行计划

**版本**: 0.0.1.1  
**创建日期**: 2026-04-16  
**目标**: 实现 OpenAI 原生工具调用 + 执行稳定性 + WebSocket 完整流式

---

## 一、当前状态分析

### 已实现
- 基础 LLM 适配器（OpenAI）
- 工具层（File/Shell/Patch）
- 执行循环（RapidExecutionLoop）
- 前端基础 UI
- 对话式交互
- WebSocket 连接管理器（基础框架）

### 存在问题
1. **工具调用方式**：依赖 Prompt 让模型手写 JSON，不稳定
2. **执行稳定性**：空响应、解析失败、工具失败时直接结束
3. **总结缺失**：工具执行后可能没有 final summary
4. **前端体验**：同步等待，长任务像"卡死"
5. **无实时反馈**：看不到 LLM 思考过程和流式输出

---

## 二、增强 MVP 目标

### 核心目标
1. OpenAI 原生 tools 调用（不再依赖 JSON 文本解析）
2. 执行循环状态机（重试/纠错/强制总结）
3. **WebSocket 完整流式**（类似 ChatGPT 体验）

### 流式输出场景

| 场景 | 内容 | 前端展示 |
|------|------|----------|
| LLM 思考 | "让我先读取文件..." | 实时逐字显示 |
| 工具调用 | tool_calls 请求 | 显示"正在执行: file" |
| 工具结果 | 执行成功/失败 | 显示结果摘要 |
| Final Summary | 任务总结 | 流式逐字显示 |

### 不在本轮范围
- 多 provider 支持
- 数据库持久化
- 复杂记忆系统

## 三、架构设计

### 3.1 三层架构

```
执行编排层 (RapidExecutionLoop)
  - 只处理统一语义：tool_calls, content
  - 通过 WebSocket 推送事件
        │
        ▼
Provider 适配层 (Adapters)
  - OpenAIAdapter: 原生 tools + 流式输出
  - 未来: ClaudeAdapter, GeminiAdapter
        │
        ▼
内部统一协议层 (Base Types)
  - LLMToolCall, LLMToolDefinition
  - LLMMessage, LLMResponse
```

### 3.2 统一协议定义

```python
class LLMToolCall(BaseModel):
    id: str
    name: str
    arguments: Dict[str, Any]

class LLMToolDefinition(BaseModel):
    name: str
    description: str
    parameters: Dict[str, Any]

class LLMMessage(BaseModel):
    role: str  # system/user/assistant/tool
    content: Optional[str]
    tool_calls: List[LLMToolCall] = []
    tool_call_id: Optional[str] = None

class LLMResponse(BaseModel):
    content: Optional[str]
    tool_calls: List[LLMToolCall] = []
    finish_reason: str
```

### 3.3 WebSocket 事件协议

```typescript
// 服务端 → 客户端事件
interface ExecutionEvents {
  // LLM 开始响应
  'llm:start': { execution_id: string }
  
  // LLM 流式输出 token
  'llm:token': { token: string }
  
  // LLM 请求工具调用
  'llm:tool_call': { 
    tool_name: string,
    args: object,
    thought: string  // LLM 的思考过程
  }
  
  // 工具执行开始
  'tool:start': { 
    tool_name: string,
    args: object 
  }
  
  // 工具执行结果
  'tool:result': { 
    tool_name: string,
    success: boolean,
    output: string,
    duration: number
  }
  
  // 执行完成
  'execution:complete': { 
    result: string,
    total_steps: number,
    duration: number 
  }
  
  // 执行失败
  'execution:error': { 
    error: string,
    step: number 
  }
}

// 客户端 → 服务端事件
interface ClientEvents {
  'execution:start': { task: string, project_id: string }
  'execution:pause': { execution_id: string }
  'execution:resume': { execution_id: string }
  'execution:stop': { execution_id: string }
}
```

### 3.4 执行循环状态机

```
START → PLANNING ←──┐
          │         │ 重试
    ┌─────┴─────┐   │
    │           │   │
  有工具     无工具 │
    │           │   │
    ▼           ▼   │
EXECUTING  FINAL_SUMMARY (流式)
    │           │
 成功/失败      │
    │           │
    └───────────┘
                │
                ▼
               DONE
```

```python
async def stream_complete_with_tools(
    self,
    messages: List[LLMMessage],
    tools: List[LLMToolDefinition] = None
) -> AsyncIterator[StreamChunk]:
    """流式输出，支持工具调用"""
    stream = await self.client.chat.completions.create(
        model=self.model,
        messages=openai_messages,
        tools=openai_tools,
        tool_choice="auto",
        stream=True
    )
    
    async for chunk in stream:
        delta = chunk.choices[0].delta
        
        # 流式输出 content
        if delta.content:
            yield StreamChunk(type="content", content=delta.content)
        
        # 工具调用（流式结束时一次性返回）
        if chunk.choices[0].finish_reason == "tool_calls":
            # 收集完整的 tool_calls
            yield StreamChunk(type="tool_calls", tool_calls=tool_calls)
```

```python
class RapidExecutionLoop:
    def __init__(self, ..., event_callback: Callable = None):
        self.event_callback = event_callback
    
    async def _emit(self, event_type: str, data: dict):
        if self.event_callback:
            await self.event_callback(event_type, data)
```

2. 在关键节点发送事件：
```python
# LLM 开始
await self._emit("llm:start", {})

# LLM 流式输出
async for token in self.llm.stream_complete_with_tools(...):
    if token.type == "content":
        await self._emit("llm:token", {"token": token.content})
    elif token.type == "tool_calls":
        await self._emit("llm:tool_call", {...})

# 工具执行
await self._emit("tool:start", {"tool_name": name})
result = await tool.execute(args)
await self._emit("tool:result", {...})
```

```python
class ConnectionManager:
    def __init__(self):
        self.connections: Dict[str, WebSocket] = {}
    
    async def connect(self, execution_id: str, websocket: WebSocket):
        await websocket.accept()
        self.connections[execution_id] = websocket
    
    async def send_event(self, execution_id: str, event: str, data: dict):
        if execution_id in self.connections:
            await self.connections[execution_id].send_json({
                "event": event,
                "data": data,
                "timestamp": time.time()
            })
```

2. WebSocket 端点：
```python
@router.websocket("/ws/execution/{execution_id}")
async def websocket_execution(
    websocket: WebSocket,
    execution_id: str
):
    await manager.connect(execution_id, websocket)
    
    try:
        while True:
            message = await websocket.receive_json()
            
            if message["event"] == "execution:start":
                # 启动执行
                asyncio.create_task(
                    run_execution_with_events(
                        execution_id,
                        message["data"],
                        event_callback=manager.send_event
                    )
                )
    except WebSocketDisconnect:
        manager.disconnect(execution_id)
```

```python
        llm=llm,
```

1. WebSocket 客户端：
```typescript
class ExecutionWebSocket {
  private ws: WebSocket;
  private handlers: Map<string, (data: any) => void>;
  
  connect(executionId: string) {
    this.ws = new WebSocket(`ws://localhost:8000/ws/execution/${executionId}`);
    
    this.ws.onmessage = (event) => {
      const { event: type, data } = JSON.parse(event.data);
      this.handlers.get(type)?.(data);
    };
  }
  
  on(event: string, handler: (data: any) => void) {
    this.handlers.set(event, handler);
  }
  
  startExecution(task: string, projectId: string) {
    this.ws.send(JSON.stringify({
      event: "execution:start",
      data: { task, project_id: projectId }
    }));
  }
}
```

2. 流式消息显示：
```typescript
const StreamingMessage = () => {
  const [content, setContent] = useState("");
  const [isStreaming, setIsStreaming] = useState(false);
  
  useEffect(() => {
    ws.on("llm:token", (data) => {
      setContent(prev => prev + data.token);
    });
    
    ws.on("llm:start", () => setIsStreaming(true));
  }, []);
  
  return (
    <div>
      {content}
      {isStreaming && <span className="cursor">▊</span>}
    </div>
  );
};
```

3. 完整流程：

```typescript
const handleSend = async () => {
  const ws = new ExecutionWebSocket();
  const executionId = generateId();
  
  ws.connect(executionId);
  
  ws.on("llm:token", (data) => {
    // 实时追加 token

    // 更新步骤结果
    updateStep(data);
  });

    // 显示最终结果
    showResult(data);
    ws.close();
  });
  
  // 启动执行
  ws.startExecution(task, projectId);
};
```
